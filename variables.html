PYTHON PROGRAMING LANGUAGE

from sympy import symbols, Eq, solve # Define the variables x, y = symbols('x y') # Define the system of equations equation1 = Eq(2*x + 3*y, 10) equation2 = Eq(4*x - 2*y, 4) # Solve the system of equations solution = solve((equation1, equation2), (x, y)) # Print the solution print("Solution for x and y:", solution)

OUTPUT

2x + 3y = 10 4x - 2y = 4
